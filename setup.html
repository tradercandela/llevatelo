-- 1. Tabla de usuarios
create table if not exists public.usuarios (
  id          uuid primary key references auth.users(id) on delete cascade,
  nombre      text not null,
  apellido    text,
  telefono    text,
  direccion   text,
  avatar      text default '😊',
  creado_en   timestamptz default now()
);
alter table public.usuarios enable row level security;
drop policy if exists "sel_usu" on public.usuarios;
create policy "sel_usu" on public.usuarios for select using (true);
drop policy if exists "ins_usu" on public.usuarios;
create policy "ins_usu" on public.usuarios for insert with check (auth.uid() = id);
drop policy if exists "upd_usu" on public.usuarios;
create policy "upd_usu" on public.usuarios for update using (auth.uid() = id);

-- 2. Tabla de pedidos
create table if not exists public.pedidos (
  id                bigserial primary key,
  usuario_id        uuid references auth.users(id),
  nombre_cliente    text not null,
  telefono_cliente  text,
  direccion_entrega text not null,
  negocio_id        int,
  negocio_nombre    text,
  items             jsonb,
  subtotal          int not null default 0,
  costo_domicilio   int default 0,
  total             int not null default 0,
  forma_pago        text default 'efectivo',
  estado            text default 'recibido',
  notas             text,
  creado_en         timestamptz default now()
);
alter table public.pedidos enable row level security;
drop policy if exists "ins_ped" on public.pedidos;
create policy "ins_ped" on public.pedidos for insert with check (true);
drop policy if exists "sel_ped" on public.pedidos;
create policy "sel_ped" on public.pedidos for select using (true);
drop policy if exists "upd_ped" on public.pedidos;
create policy "upd_ped" on public.pedidos for update using (true);

-- 3. Índices
create index if not exists idx_ped_usuario on public.pedidos(usuario_id);
create index if not exists idx_ped_estado  on public.pedidos(estado);
create index if not exists idx_ped_fecha   on public.pedidos(creado_en desc);

2
Desactivar confirmación de correo
Para que el login funcione sin tener que confirmar emails.
→En Supabase, ve a Authentication → Providers
→Busca Email y haz clic para expandir
→Desactiva el toggle "Confirm email"
→Clic en Save
ℹ️ Esto permite que cualquier usuario entre de inmediato sin esperar un correo de verificación. Perfecto para el MVP.
